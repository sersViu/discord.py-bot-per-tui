            command_prefix="!",
            help_command=None,

            application_id=config["application-id"]


       )

    async def setup_hook(self):
        for file in Path("cogs").glob("*.py"):

            print("loaded extention:", file.name)

    async def on_ready(self):
        logging.info(f"{self.user.name} has connected to Discord!")
